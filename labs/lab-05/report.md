#### Makefile
```
all: dynamic_block static_block
clean:
	rm *.o *.so *.a
dynamic_block: program.o libblock.so
	cc program.o libblock.so -o dynamic_block -Wl,-rpath .
static_block: program.o libblock.a
	cc program.o libblock.a -o static_block
libblock.so: block.o
	cc -shared -o libblock.so block.o
block.o: block.c
	cc -fPIC block.c -o block.o
libblock.a: block.o
	ar qc libblock.a block.o
block.o: block.c
	cc -c block.c -o block.o
block.c: ../headers/block.h
program.o: program.c
	cc -c ../program.c -o program.o
program.c: ../headers/block.h
```
#### CMakeLists.txt
```
cmake_minimum_required(VERSION 3.10)
project(Block)

add_library(dynamicblock SHARED block.c ../headers/block.h)
add_library(staticblock STATIC block.c ../headers/block.h)

add_executable(dynamic_block_cmake ../program.c)
target_link_libraries(dynamic_block_cmake dynamicblock)

add_executable(static_block_cmake ../program.c)
target_link_libraries(static_block_cmake staticblock)
```
